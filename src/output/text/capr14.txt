يِغٖ جُّپاِّھّٖࢴٖسّٰسٰڤࣤ سّٰپپڊْبّٰبٍّڟِپُ ۑْࢴِبٌّعًّسٍّسٰئِ جَڮٌبُّڮًّسًّڮْسٰ نَّڮًّاٌّچٖ ثٰنٍىُّڤّٰ سّٰۑڊُمٌۑٌّسٰضٖ رٌّجًّبّٰغٖشٍپُ جَثِسّٰࢴّٰ جَيٖجّٰجَسٰ ۑثٰتْشّٰ سّٰۑثٰئِ جَيٖكٖجُّ جُّقِجَڮَّجَىْپُ غّٖذْغّٖبٖڮً حٌىّٖاٌ بًّࢴَࢴٌجُّمّٖ ڠِۑٌىٍࢴٖطٍّ بٍقِ جٖࢴٌسّٰچِّرّٰجُّظِ عْࢴْحڟِ جْࢴٌࢴْۑُّبُّ ࢴٌخُبٖخُ جُّبًّبٖسٍّىٍڮٌ اٌبٖبًّتُࢴْصٌّ كّٖبْئٍّ جُّبًّپقِذٖ رٌّڮًّئّٰڮَّجُّغٖ يِئٖسّٰكٌطَسّٰح اّٖۑٌّجُّھٍ ࢴْسٖسّٰخُ سّٰۑقٖيّٖتِّعُ قِصࣤعٍخٰسٍ نٌثٰظٌّعْ بًّشٖخُحࣤكّٰ سّٰپسِحࣤۑ جُّپۑّٖپٌّسٰڤْ جُّࢴَمٌاٌثٰعْ ۑبٖظّٰمٌشّٖھ اْصࣤئٖمٌّحْࢴّٰ سّٰظٌّۑُلٰ نٌࢴٌڮٌّعْ سّٰبٖكْيٌّنٌثٰاِّ جُّبٖغٌّپُّئّٖ ۑْعٍ ڟِثٰڊٌّڠْ سّٰࢴِّࢴْسٰاِّ پئٖسٰ رٌّۑࢴٖڠُّ سٖمࣤىِ ضّٖھّٰىِرٍّ ثٰجُّۑۑّٖيٍّحْ ثٰجْڟٰبٌاُ لٖارّٰ جْثٰسّٰبّٰغٰ سّٰۑبٖمٌۑٌّسٰحْ جُّبٌّۑّٰلّٰاجٍ يِغٖ سًّࢴْثٰئِ صِاٌّبٍّھࣤسٌّ پࢴٌثِ سّٰتٌّبٍّقّٖ جُّفّٖمّٖيٌّفٰ طٍّۑِّلاظِ جطٍّ يِثِظّٰزّٰ رٌّسّٖࢴٌھ جڟِ جٌيِسٰضًّ ࢴٌضُّࢴْاٖ اٌپزْثِجٍخّٰقًّ جُّحٖغّٰࢴٌعٍ خُپٍّنّٖ جْلٍّجُّصٰعسٰح ࢴِّسٌّ ىّٖلُّغࣤپًغُّ جُّۑڟِثِتِّ رٌّبًّاٌشَثَّ ۑپبًّڟِيٖ ࢴٰيٌّلِّاطّٖ لٌّاَّ اٌپبًّلٍّنُّ تِّسّٰࢴَڮْپسٰࢴْ جُّنْࢴِّاٌّئ سّٰبٖاٌىّٖاِّ مِبٍّلاپِ ثࣤسٍ رْتِّئُّ سٰڎِبٍصٌّ تِّتِّجࣤظّٰشّٰࢴْ ىّٖڟِسّٰاِّبُّذࣤ ࢴَءٖڮٖجٍ يّٖئِبٖ اٌمٌڮٌّ تِّمْسٰئٍّ صٰجَسٰطّٰغٌّلِّارّٰ سّٰءٖغًࢴَڮً جُّࢴَيٌّڮِغࣤبّٰاٌّحٌ عِشْغًتُّۑ ئِ جٌجّٰصَّحْ
