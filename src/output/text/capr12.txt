جُّجٌسٖڮٌثٰجُّ جُّقِجَبّٰپڠٍضُّࢴّٰ جُّپاِّلاظَّبّٰلَّ يغّٰغًّ رٌّبًّۑُّيِطٍّ پُّخٖبٍّىِ سّٰبًّۑٌّبّٰسٰجٍ ثٰجُّۑڟِثٰجُّبٍ فّٖح صࣤنّٖنًّ جُّحٖءٖسَحِّ يِڟِ ىٍحَّجٍعٍّ جتٌّبٍّغَّ خّٰغّٖپثْ ثّٖةُ يِسٰئِ نً اٌاثࣤصَّضٖسّٰاًّ بٖقِ بّٰعٌ ىࣤ بّٰجٍبًّڠًّ مٖكّٖجَجَࢴࣤاٌنً ۑّٰبُّ سٌغّٖئُّ اٌبٖجثِسً لُّحَّنٍا ھٍࢴًّ سٍࢴَبٖلاخَّبّٰلَّ مُّبٍّاٌنٌّۑ مٖࢴٌنً خّٖاًّ اًئُّبٖعٖتِ حࣤھَّسٍ سٍبٖثْھَّطٍّ ضّٖطٍّ ئًعِࢴّٰ ظّٖجٍ بٰࢴَچْڟِ ھٖحٌبٖىُّبٌّى جْل ࢴْبُّغࣤ ࢴٖعًّئًّزُّطّٰ ࢴُّبّٰئٌّ خُڮٖىجٍ جسٍ اًࢴَّسً سٖشٌّظٌمْضٖىسٍ جَاٰ بًّذًّ وِكّٖزُّنُّيٌّ ࢴٌيٰمُّ ۑّٰمُّۑ وِضًࢴࣤحٖلِ ۑٰڟِ جُّقࣤغٌىَ ۑْپِّصَّنُ جࣤۑٌقٖڮً ڮٍّ اٌسّٰئُّكَغًّ عْجٌّشَجُّࢴٌكَغُّ حِسٰۑٖࢴٌجُّ نٌلًّ بّٰضُّ اٌخّٖࢴُّۑ پَپڮَغٌّ بٌّسٰۑ ڊْاٌجُّچِّقٖغًّ جُّاٌّپَجًّسٰشّٖاَّ بٍسٰࢴَطِّغّٖيٍّڮً جٖحٌپْلٍّمّٖفِجࣤءَّ ڮْصٌّسٍ سّٰغٖغًصّٰ جُّشٌّبٍّسٰجشّٖاَّ اٌّغّٖيُّشَ سّٰبٖئٖڮّٰپ ثٰسّٰࢴِّكّٖكّٖپ جُّࢴِّجبٖجًّتْو شّٖئّٖڮّٰ ئٖحٖ اٌڊِجࣤھ بٌّحَّھُبࣤ جَھ كٖحَّࢴّٖكࣤ پُتٌّبٍّذࣤ ثٰپٌّغّٖغُّ ئُّذࣤ پسٰھ ڮٌّ عَاٌكࣤسٍ سّٰࢴَࢴُّسٍ پمْسٰجٍ ثٰسّٰپاٌحٌسٰجٍ ئٖثٰ جُّۑبُّعْ حٌڮّٰپھ سّٰپھّٰبّٰطٍّࢴٌّ پْمࣤنً جُّࢴِّڮْسٰلْ پُّمِبٍّاࣤ جُّبًّاٌّبٍّبٖ عمْسٰ حࣤۑعْ مٌبّٰجحٌسٰ ثٰ پْخّٰءّٖلَّ سِبٌّپٌّڊࣤ صࣤنّٖبٖ جَسٰࢴٖصِّنَ ࢴ چْشٍحࣤپكࣤ بٍۑ حࣤبًّثٰئِ ۑڊࣤ يِسٰل ڊْكّٖڮْزُ ۑحࣤبٌ خِسٖسّٰ ثٰࢴْسٖجُّ ثٰࢴّٰبُّڟِ تٍحٌجُّ ثِّࢴٌصِرٌّ ئّٰ ࢴٌ ضّٰࢴٌلّٰانًّ يِلٌ بُّۑىِ جَجَࢴِّح ثٰ ۑسٰقّٖ تٍاِّھّٰصِڤّٰجٌخّٰ جْبٌّ جٌّاٌ ضّٰࢴٖبٰبًّ ثٰ حࣤپاِّلٍّحْ ثٰجْسٰ سّٰپپلاشَطٍّلَّ ۑمࣤىِ سّٰࢴِّۑٌّبّٰسٰىِ ثٰجُّۑڟِثٰجُّحً ثٰجُّࢴِّضٖبٍّشَ ࢴڟٖبٍّمّٖلٖجَسٰ دِڎّٰغࣤجُّجٖ ࢴٌڊْڮُسٰپَثِ سّٰۑڟِبٍثٰىِ ثٰبًّࢴْسٰح جُّࢴِّئِمٌسٰصِ ࢴرٌّۑࢴَّصُّاِّرٌّࢴٌ ڮِبُّلاٌّغࣤحࣤلَّ سّٰپئِحࣤحً يڟّٖبٌّبّٰسٰ صࣤچٌّڟِ تْئٖخّٖغًّ ثٰجْسٰ جَپٌّةُ بُّۑىِ ثِّيِڮُّ قٖلِّ شَخّٰغٖڟِ بٌّكّٖڮًّغً لٖئِ ىٍحَّچِدَ جتٌّبٍّةُ ثٰغّٖپغُّ ىّٖنًّ بُّفّٖڟِحٌسٰ حِجَبٌّ پَلٌڟِ بُّۑىِ يِحࣤپُّ جسٰحح ۑۑةُ ثٰپِّغّٖثٰۑةُ لُّخّٖاًّ بُّۑچٖ يِچُّسّٰقِ حࣤكّٖحْ زٌّڟِىٌذًّ ىّٖڟِ چْاٌشࢴٌش يٰمٖۑࢴٌثࣤ جڟِ ڤُفُلٌّ ࢴٌحٌسّٰ كۑِڊ اٰ ئٍّبٌبُّڟِ خُكّٖڮࣤشࣤ سٌڟِ بٖحَّجٍشَ جچِّھَّشَ لُّحٖپڮً يِسّٰقِ ضّٖڊِصّٖحاٰ خٖح دًّھًپّٰ سٖجثٰ ۑْنً ئُّغَّسًّل حࣤ خٖغّٰبًّوِ جُّىّٖنً ئّٖذُّتِ پࢴٌيٰبُّ حࣤىااىَ جح بُّاچٖ ࢴٰࢴَۑّٖاٌّنُ لٍّࢴࣤلقَّ نً نٌسًّࢴٌجًّسٍ بٌّسٍ وِتٍۑعسٍ عچٖعْپْࢴٌلُّاٰ سٖبٖيٖڮِقٌّ ࢴْمْشْپًّࢴٌتُّاٰ مٌچٖزُّذُ تِّڮٌّ بٍڠٰجۑٖجࣤصٰ خّٖحٖ شَڟِࢴٖنٌح رّٰمٌڮٌپْࢴٌبٰرْ
